# Build stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# copy only project file first
COPY FP.csproj ./
RUN dotnet restore FP.csproj

# copy everything else
COPY . ./
RUN dotnet publish FP.csproj -c Release -o /app

# Runtime stage
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app

# Copy published files from build stage
COPY --from=build /app ./

# Make sure SQLite database file can be created
RUN mkdir -p /app/Data && chmod -R 777 /app/Data

# Expose port
EXPOSE 80

# Start the application
ENTRYPOINT ["dotnet", "FP.dll"]
